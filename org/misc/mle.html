Y <- rnorm(1000, mean=10, sd=4)

norm_log_likelyhood <- function(x, y) {
  -sum(1/2*(y-x[1])^2/x[2] + 1/2*log(x[2]))
}

norm_log_likelyhood(c(10,2),Y)

optim(c(10,2), norm_log_likelyhood, y=Y, control=list(fnscale=-1))
